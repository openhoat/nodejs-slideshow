Modules

  • Node fournit quelques variables globales
    • module, process, mais pas window
  • Et un mécanisme pour rendre les choses modulaires
    • Inspiration pythonesque ☺
// lib/mymodule.js
module.exports = {
  foo: function(){
    return 'bar';
  }
};
    
// app.js
var myModule = require('./lib/mymodule');
myModule.foo();
