Simple db helper 

var mongoose = require('mongoose')
  , db = { connection:null,
    connect:function () {
      db.connection = mongoose.connect('mongodb://localhost:27017/db');
    }, close:function () {
      if (db.connection !== null) {
        mongoose.disconnect();
        db.connection = null;
        return true;
      } else return false;
    }
  };
db.connect();
process.on('exit', function (code) {
  var result = db.close();
  console.log('disconnecting db : ', result);
});
